示例
code - datebox.js
var layout = new Ele.Layout();
var db = new Ele.DateBox();
var btn = new Ele.Button("按钮");
layout.ele.onclick = function(){
db.showBelowLeft(layout.ele);
}
db.setSelectUpdateHandler(function(res){
alert("您选择了："+db.getSelectDateString());
});
layout.add(btn);
layout.setContainerById("main");
